# Notification Service Deployment
apiVersion: apps/v1
kind: Deployment
metadata:
  name: notification-service
  namespace: avgc-tender-system
  labels:
    tier: backend
spec:
  replicas: 2
  selector:
    matchLabels:
      app: notification-service
      tier: backend
  template:
    metadata:
      labels:
        app: notification-service
        tier: backend
      annotations:
        prometheus.io/scrape: "true"
        prometheus.io/port: "9090"
        prometheus.io/path: "/metrics"
    spec:
      serviceAccountName: avgc-app-sa
      containers:
      - name: notification-service
        image: avgc/notification-service:latest
        ports:
        - containerPort: 3000
        - containerPort: 9090  # Metrics port
        env:
        - name: DATABASE_URL
          valueFrom:
            secretKeyRef:
              name: db-secrets
              key: auth-db-url
        - name: RABBITMQ_URL
          value: "amqp://rabbitmq-service:5672"
        - name: REDIS_URL
          value: "redis://redis-service:6379"
        - name: NODE_ENV
          value: "production"
        - name: EMAIL_HOST
          value: "smtp.gmail.com"
        - name: EMAIL_PORT
          value: "587"
        - name: EMAIL_USER
          valueFrom:
            secretKeyRef:
              name: email-secrets
              key: email-user
        - name: EMAIL_PASS
          valueFrom:
            secretKeyRef:
              name: email-secrets
              key: email-pass
        - name: SMS_API_KEY
          valueFrom:
            secretKeyRef:
              name: sms-secrets
              key: api-key
        resources:
          requests:
            memory: "256Mi"
            cpu: "200m"
          limits:
            memory: "512Mi"
            cpu: "400m"
        livenessProbe:
          httpGet:
            path: /health
            port: 3000
          initialDelaySeconds: 30
          periodSeconds: 10
        readinessProbe:
          httpGet:
            path: /ready
            port: 3000
          initialDelaySeconds: 10
          periodSeconds: 5

---
# Notification Service
apiVersion: v1
kind: Service
metadata:
  name: notification-service
  namespace: avgc-tender-system
spec:
  selector:
    app: notification-service
  ports:
  - name: http
    protocol: TCP
    port: 3000
    targetPort: 3000
  - name: metrics
    protocol: TCP
    port: 9090
    targetPort: 9090
  type: ClusterIP

---
